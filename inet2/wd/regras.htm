Regras para cálculo do indicador de atraso da ANAC na analise dos processos (resDelayANAC):


        '--------------------------------------------------------------------------
        ' resDelayANAC	- Indica atraso da ANAC na analise dos processos
        If m_oAIRData.ANAC = 0 Then
            resDelayANAC = RES_NONE

        ' CRITICAL
        ElseIf m_oAIRData.ANAC >  And _
           ( m_oAIRData.ANACDelay >  Or _
             m_oAIRData.ANACMaxDays >  Or _ 
           ( m_oAIRData.ANAC >  And _
             m_oAIRData.ANAC60d/m_oAIRData.ANAC >  ) Or _
           ( m_oAIRData.ANAC >  And _
             m_oAIRData.ANAC30d/m_oAIRData.ANAC >  ) ) Then
        resDelayANAC = RES_CRITICO


        ' RES_RUIM
        ElseIf m_oAIRData.ANAC >  And _
               ( m_oAIRData.ANACDelay >  Or _
                 m_oAIRData.ANACMaxDays >  Or _ 
                 ( m_oAIRData.ANAC >  And _
                   m_oAIRData.ANAC60d/m_oAIRData.ANAC >  ) Or _
                 ( m_oAIRData.ANAC >  And _
                   m_oAIRData.ANAC30d/m_oAIRData.ANAC >  ) ) Then
                resDelayANAC = RES_RUIM

        ' RES_ALERTA
        ElseIf m_oAIRData.ANAC >  And _
               ( m_oAIRData.ANACDelay >  Or _
                 m_oAIRData.ANACMaxDays >  Or _ 
                 ( m_oAIRData.ANAC >  And _
                   m_oAIRData.ANAC60d/m_oAIRData.ANAC >  ) Or _
                 ( m_oAIRData.ANAC >  And _
                   m_oAIRData.ANAC30d/m_oAIRData.ANAC >  ) ) Then
                resDelayANAC = RES_ALERTA

        ' RES_BOM
        ElseIf m_oAIRData.ANAC >  And _
               ( m_oAIRData.ANACDelay >  Or _
                 m_oAIRData.ANACMaxDays >  Or _ 
                 ( m_oAIRData.ANAC >  And _
                   m_oAIRData.ANAC60d/m_oAIRData.ANAC >  ) Or _
                 ( m_oAIRData.ANAC >  And _
                   m_oAIRData.ANAC30d/m_oAIRData.ANAC >  ) ) Then
                resDelayANAC = RES_BOM

        ' RES_OTIMO
        ElseIf m_oAIRData.ANAC >  And _
               ( m_oAIRData.ANACDelay >  Or _
                 m_oAIRData.ANACMaxDays >  Or _ 
                 ( m_oAIRData.ANAC >  And _
                   m_oAIRData.ANAC60d/m_oAIRData.ANAC >  ) Or _
                 ( m_oAIRData.ANAC >  And _
                   m_oAIRData.ANAC30d/m_oAIRData.ANAC >  ) ) Then
                resDelayANAC = RES_OTIMO

        ' RES_NONE
        Else
            resDelayANAC = RES_NONE
        End If
